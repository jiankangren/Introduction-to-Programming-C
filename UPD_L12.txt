//ex1
#include <stdio.h>

int factorial (int x){
	if (x == 0){
		return 1;
	} else {
		return factorial (x-1) * x;
	}
}

int main (){

	int i, n = 10;
	int arrFact[n];

	for (i = 0; i < n; i++){
		arrFact[i] = factorial(i);
		printf ("%i \n", arrFact[i]);
	}
	return 0;
}

#include <stdio.h>
#include <stdlib.h>

int random (){
 	return rand()% 50 + 51;
}

int min (int arr[], int n){
	int minVal = arr[0], i;

	for (i = 0; i < n; i++){
		if (minVal < arr[i]){
			minVal = arr[i];
		}
	}
	return minVal;
}

int max (int arr[], int n){
	int maxVal = arr[0], i;

	for (i = 0; i < n; i++){
		if (maxVal > arr[i]){
			maxVal = arr[i];
		}
	}
	return maxVal;
}

int mean (int arr[], int n){
	int meanVal = 0, i;

	for (i = 0; i < n; i++){
		meanVal += arr[i];
	}
	return meanVal / n;
}

int main (){

	int i, n = 20;
	int arrFact[n];

	FILE * doodle;

	doodle = fopen ("doodle.txt", "w");

	if (doodle == NULL){
		printf ("Can't open file");
		return 1;
	}

	printf ("File opened successfully");

	for (i = 0; i < n; i++){
		arrFact[i] = random();
		fprintf (doodle,"%i \n", arrFact[i]);
	}

	int minVal = min(arrFact, n);
	int maxVal = max(arrFact, n);
	int meanVal = mean(arrFact, n);

	fprintf (doodle, "max is %i \n"
			"min is %i \n"
			"mean is %i \n", minVal, maxVal, meanVal);

	fclose (doodle);
	return 0;
}
