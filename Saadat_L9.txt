#include <stdio.h>

int main (){

	setvbuf(stdout, NULL, _IONBF, 0);

	int n, i;

	printf("Enter value here: \n");
	scanf ("%i", &n);

	printf("This value is divisible by:");

	for (i = 2; i < n; i++){
		if (n % i == 0){
			printf(" %i,", i);
		}
	}

	return 0;
}
